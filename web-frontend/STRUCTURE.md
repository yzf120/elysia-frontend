[//]: # (### Vite 配置 &#40;vite.config.js&#41;)

[//]: # ()
[//]: # (```javascript)

[//]: # ({)

[//]: # (  root: 'src',              // 源码目录)

[//]: # (  server: {)

[//]: # (    port: 3000,             // 开发服务器端口)

[//]: # (    open: true,             // 自动打开浏览器)

[//]: # (    proxy: {                // API 代理配置)

[//]: # (      '/api': {)

[//]: # (        target: 'http://localhost:8001',)

[//]: # (        changeOrigin: true)

[//]: # (      })

[//]: # (    })

[//]: # (  },)

[//]: # (  build: {)

[//]: # (    outDir: '../dist',      // 构建输出目录)

[//]: # (    emptyOutDir: true       // 构建前清空目录)

[//]: # (  })

[//]: # (})

[//]: # (```)
